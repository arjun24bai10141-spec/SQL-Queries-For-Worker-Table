	WORKER_ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	FIRST_NAME CHAR(25),
	LAST_NAME CHAR(25),
	SALARY INT(15),
	JOINING_DATE DATETIME,

CREATE TABLE Bonus (
	WORKER_REF_ID INT,
	BONUS_AMOUNT INT(10),
	BONUS_DATE DATETIME,
	FOREIGN KEY (WORKER_REF_ID)
		REFERENCES Worker(WORKER_ID)
        ON DELETE CASCADE
);

-- Fetch first name using alias WORKER_NAME

SELECT FIRST_NAME AS WORKER_NAME
FROM Worker;



-- Show unique department names

SELECT DISTINCT DEPARTMENT
FROM Worker;



-- Find details of employees named Vipul and Satish

SELECT *
FROM Worker
WHERE FIRST_NAME IN ('Vipul', 'Satish');



-- Find employees whose first name contains 'a'

SELECT *
FROM Worker
WHERE FIRST_NAME LIKE '%a%';



-- Find employees whose first name ends with 'a'

SELECT *
FROM Worker
WHERE FIRST_NAME LIKE '%a';



-- Show admin employees with salary between 100000 and 500000

SELECT *
FROM Worker
WHERE DEPARTMENT = 'Admin'
AND SALARY BETWEEN 100000 AND 500000;



-- Count employees working in Admin department

SELECT COUNT(*) AS TOTAL_EMPLOYEES
FROM Worker
WHERE DEPARTMENT = 'Admin';



-- Show the second highest salary from the table

SELECT MAX(SALARY) AS SECOND_HIGHEST_SALARY
FROM Worker
WHERE SALARY < (
    SELECT MAX(SALARY)
    FROM Worker
);



-- Show departments having less than three employees

SELECT DEPARTMENT
FROM Worker
GROUP BY DEPARTMENT
HAVING COUNT(*) < 3;



-- Show all departments along with total number of employees

SELECT DEPARTMENT, COUNT(*) AS TOTAL_EMPLOYEES
FROM Worker
GROUP BY DEPARTMENT;



-- Print the names of employees having the highest salary in each department

SELECT FIRST_NAME, DEPARTMENT, SALARY
FROM Worker W1
WHERE SALARY = (
    SELECT MAX(SALARY)
    FROM Worker W2
    WHERE W1.DEPARTMENT = W2.DEPARTMENT
);
