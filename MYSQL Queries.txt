-- Fetch first name using alias WORKER_NAME

SELECT FIRST_NAME AS WORKER_NAME
FROM Worker;



-- Show unique department names

SELECT DISTINCT DEPARTMENT
FROM Worker;



-- Find details of employees named Vipul and Satish

SELECT *
FROM Worker
WHERE FIRST_NAME IN ('Vipul', 'Satish');



-- Find employees whose first name contains 'a'

SELECT *
FROM Worker
WHERE FIRST_NAME LIKE '%a%';



-- Find employees whose first name ends with 'a'

SELECT *
FROM Worker
WHERE FIRST_NAME LIKE '%a';



-- Show admin employees with salary between 100000 and 500000

SELECT *
FROM Worker
WHERE DEPARTMENT = 'Admin'
AND SALARY BETWEEN 100000 AND 500000;



-- Count employees working in Admin department

SELECT COUNT(*) AS TOTAL_EMPLOYEES
FROM Worker
WHERE DEPARTMENT = 'Admin';
